JSON Messages:

Question:
{ Desc:Str, Topic:Str, Id:int, Diff:int, Tests:[Str] }

Test:
{ TestId:int, TestName:Str, Released:int, QIds:[int] }

Answer:
{ TestId:int, QId:int, Text:Str }

AddQ:
Request { Type:AddQ, Desc:Str, Topic:Str, Diff:int, Tests:[Str] }
Reply 	{ Type:AddQ, Error:Str, {Question} }

SearchQ:
Request { Type:SearchQ, Topic:Str, Diffs:[int], Keys:[Str] }
Reply	{ Type:SearchQ, Error:Str, Questions:[{Question}] }

AddTest:
Request { Type:AddTest, TestName:Str, Release:int, QIds:[] }
Reply	{ Type:AddTest, Error:Str, Test:{Test}}

GetTest:
Request { Type:GetTest, TestIds:[], Release:int }
Reply	{ Type:GetTest, Error:Str, Tests:[{Test}] }

SubmitTest:
Request { Type:SubmitExam, Answers:[{Answer}] }
